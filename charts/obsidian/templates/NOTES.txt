1. Get the application URL by running these commands:
{{- if .Values.ingress.enabled }}
{{- range $host := .Values.ingress.hosts }}
  {{- range .paths }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}{{ .path }}
  {{- end }}
{{- end }}
{{- else if contains "NodePort" .Values.service.type }}
  export NODE_PORT_HTTP=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "obsidian.fullname" . }})
  export NODE_PORT_HTTPS=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[1].nodePort}" services {{ include "obsidian.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo "HTTP URL:  http://$NODE_IP:$NODE_PORT_HTTP"
  echo "HTTPS URL: https://$NODE_IP:$NODE_PORT_HTTPS (recommended)"
{{- else if contains "LoadBalancer" .Values.service.type }}
     NOTE: It may take a few minutes for the LoadBalancer IP to be available.
           You can watch the status of by running 'kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "obsidian.fullname" . }}'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "obsidian.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo "HTTP URL:  http://$SERVICE_IP:{{ .Values.service.httpPort }}"
  echo "HTTPS URL: https://$SERVICE_IP:{{ .Values.service.httpsPort }} (recommended)"
{{- else if contains "ClusterIP" .Values.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "obsidian.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT_HTTPS=$(kubectl get pod --namespace {{ .Release.Namespace }} $POD_NAME -o jsonpath="{.spec.containers[0].ports[1].containerPort}")
  echo "Visit https://127.0.0.1:3001 to access Obsidian"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 3001:$CONTAINER_PORT_HTTPS
{{- end }}

2. Important Notes:
   - HTTPS (port {{ .Values.service.httpsPort }}) is REQUIRED for full functionality (WebCodecs support)
   - The container uses a self-signed certificate by default
   - HTTP (port {{ .Values.service.httpPort }}) has limited functionality

{{- if not .Values.auth.enabled }}
3. Security Warning:
   - No authentication is currently configured
   - Set auth.enabled=true and configure auth.username/auth.password for basic auth
   - For production use, enable authentication or use a reverse proxy with robust authentication
{{- end }}

{{- if .Values.persistence.enabled }}

   - Volume size: {{ .Values.persistence.size }}
{{- else }}
4. Warning: Persistence is disabled

   - Enable persistence in production environments
{{- end }}
